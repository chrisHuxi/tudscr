\def\castseries#1#2{
  \begingroup
    \edef\@tempa{\str{#1}}
    \edef\@tempb{r}
    \ifx\@tempa\@tempb\relax
      \edef\@tempa{m}
    \else
      \edef\@tempb{s}
      \ifx\@tempa\@tempb\relax
        \edef\@tempa{sb}
      \else
        \edef\@tempb{x}
        \ifx\@tempa\@tempb\relax
          \edef\@tempa{eb}
        \fi
      \fi
    \fi
    \edef\@tempb{
      \endgroup
      \noexpand\resetstr{#2}{\@tempa}
    }
  \@tempb
}
\def\lowerstr#1#2{
  \begingroup
    \edef\@tempa{\lowercase{\def\noexpand\@tempa{\str{#1}}}}
    \@tempa
    \edef\@tempb{
      \endgroup
      \noexpand\resetstr{#2}{\@tempa}
    }
  \@tempb
}
\makeatother
